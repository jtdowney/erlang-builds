Erlang/OTP APT repository

Signed Debian/Ubuntu packages for Erlang/OTP (amd64 + arm64).

curl -fsSL __BASE_URL__/key.gpg | sudo gpg --dearmor -o /usr/share/keyrings/erlang.gpg
echo "deb [signed-by=/usr/share/keyrings/erlang.gpg] __BASE_URL__ $(. /etc/os-release; echo $VERSION_CODENAME) main" \
  | sudo tee /etc/apt/sources.list.d/erlang.list
sudo tee /etc/apt/preferences.d/erlang.pref <<'EOF'
Package: erlang*
Pin: origin jtdowney.github.io
Pin-Priority: 1001
EOF
sudo apt update
sudo apt install erlang

Optional components: erlang-dev, erlang-wx, erlang-odbc.
